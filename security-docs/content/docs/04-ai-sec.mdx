// ✅ AI エージェントのガードレール実装
interface AgentAction {
  type: 'read' | 'write' | 'delete' | 'external_api';
  resource: string;
  riskLevel: 'low' | 'medium' | 'high' | 'critical';
}

async function executeWithGuardrails(action: AgentAction): Promise<void> {
  // 1. 最小特権: 許可されたアクションのみ実行
  if (!ALLOWED_ACTIONS.includes(action.type)) {
    throw new SecurityError('Unauthorized action type');
  }

  // 2. ユーザーコンテキストで実行（特権ID使用禁止）
  const userCtx = await getUserOAuthContext();

  // 3. Human-in-the-loop: 高リスクアクションは承認必須
  if (action.riskLevel === 'high' || action.riskLevel === 'critical') {
    const approved = await requestHumanApproval(action);
    if (!approved) throw new SecurityError('Action rejected');
  }

  await executeAction(action, userCtx);
}
