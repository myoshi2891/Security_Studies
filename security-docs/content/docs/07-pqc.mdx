// ✅ Crypto-Agility: アルゴリズムをハードコードしない設計
interface CryptoConfig {
  kex: 'X25519' | 'ML-KEM-768' | 'X25519_ML-KEM-768';  // 鍵交換
  sig: 'Ed25519' | 'ML-DSA-65' | 'Ed25519_ML-DSA-65';   // 署名
}

// 設定ファイルから動的に読み込み
const config: CryptoConfig = loadConfig('crypto.yml');

// アルゴリズム切り替えはコード変更不要
const session = await tls.connect({
  host: 'api.example.com',
  keyExchange: config.kex,   // ← ハードコードしない
  signature: config.sig,
});
