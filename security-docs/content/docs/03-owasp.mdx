      サーバーに内部リソースへのアクセスを強制するSSRF攻撃シナリオ：

```bash title="SSRF Attack Vector"
# 攻撃者が画像インポートURLを悪用
# 正常: https://app.com/api/v1/load-image?url=https://example.com/photo.jpg
# 攻撃: url=http://169.254.169.254/latest/meta-data/iam/security-credentials/

# 対策: IMDSv2の強制適用（AWSの場合）
aws ec2 modify-instance-metadata-options \
  --instance-id i-1234567890abcdef0 \
  --http-tokens required \
  --http-endpoint enabled
```
    <Callout type="error" title="対策ポイント">URLホワイトリストの実装 + IMDSv2の強制適用で、内部メタデータへのアクセスを防止してください。</Callout>
  </Tab>

// ✅ セキュアな例外処理パターン
export async function handleRequest(req: Request): Promise<Response> {
  try {
    const result = await processBusinessLogic(req);
    return Response.json(result);
  } catch (error) {
    // サーバーログに詳細を記録（PIIはスクラブ）
    logger.error('Request failed', {
      requestId: req.headers.get('x-request-id'),
      error: sanitize(error),  // APIキー・PII除去
    });
    // ユーザーには汎用メッセージのみ返却
    return Response.json(
      { message: '予期せぬエラーが発生しました。' },
      { status: 500 }
    );
  }
}
