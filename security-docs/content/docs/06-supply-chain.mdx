```bash title="supply-chain-verify.sh"
# ステップ1: 署名付きSBOMの検証
cosign verify <spdx_sbom_file>

# ステップ2: コンテナイメージの署名検証
cosign verify $IMAGE \
  --certificate-identity=https://github.com/my-org/my-repo/.github/workflows/build.yml@refs/heads/main \
  --certificate-oidc-issuer=https://token.actions.githubusercontent.com

# ステップ3: SLSA来歴（Provenance）の検証
slsa-verifier verify-image "$IMAGE" \
  --provenance-path provenance.json \
  --source-uri github.com/my-repo/my-application \
  --source-tag v1.2.0
# → PASSED: Verified SLSA provenance
```
